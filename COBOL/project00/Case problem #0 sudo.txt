case problem0 sudo
author nick houser

Variable Definitions
	ALPHA-NUMERIC- I-REC, I-ID, I-NAME, I-LNAME, IFNAME, I-INIT, O-DATE, 
	O-PCTR, 0-GPA, O-START-SALARY, O-ID, O-LNAME, O-FNAME, O-SCTR, EOF = "NO"
	NUMERIC I-GPA, I-START-SALARY, I-DATE, C-SCTR=0, C-PCTR=0
	
MAIN
	CALL INIT
	WHILE EOF == "NO"
		CALL MAINLINE
	CALL CLOSING

INIT
	OPEN INPUT FILE
	OPEN OUTPUT FILE
	REQUEST SYSTEM DATE
	GET I-DATE
	O-DATE = I-DATE
	CALL READ
	CALL HEADINGS

MAINLINE
	CALL CALCS
	CALL OUTPUT
	CALL READ

CALCS
	C-SCTR +=1
	
OUTPUT
	O-ID = I-ID
	O-LNAME = I=LNAME
	O-FNAME = I-FNAME
	O-GPA = I-GPA
	O-START-SALARY = I-START-SALARY
	PRINT DETAIL LINE
	PRINT BLANK LINE
	IF END OF PAGE
		CALL HEADINGS
		
CLOSING
	O-SCTR = C-SCTR
	PRINT BLANK LINE
	PRINT TOTAL LINE
	CLOSE INPUT FILE
	CLOSE OUTPUT FILE

READ
	READ RECORD: I-REC, I-ID, I-NAME, I-LNAME, I-FNAME, I-INIT, I-GPA, I-START-SALARY
		IF END OF FILE
			EOF = "YES"
						
HEADINGS
	C-PCTR += 1
	O=PCTR = C-PCTR
	PRINT TITLE LINE
	PRINT BLANK LINE
	PRINT COLUMN HEADING LINE 1
	PRINT COLUMN HEADING LINE 2
	PRINT BLANK LINE
