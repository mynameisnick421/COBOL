Case Problem 01
Author Nick Houser

INPUT VARIABLE DEFINITIONS 
	ALPHA-NUMERIC--- I-PAINT-EST-NO, I-PAINT-DATE,
	NUMERIC--- I-PAINT-YY, I-PAINT-MM, I-PAINT-DD
		I-PAINT-WALL-SQ-FT, I-PAINT-DOOR-SQ-FT, 
		I-PAINT-PRICE-GAL, I-DATE
	
CALCS VARIABLE DEFINITIONS
	ALPHA-NUMERIC--- EOF = "FALSE"
	NUMERIC--- C-TOTAL-SQ-FT, C-GALLONS-NEEDED,
		C-PAINT-EST, C-LABOR-EST, C-TOTAL-EST, C-PCTR =0,
		C-GAL-NEEDED-GT =0, C-PAINT-EST-GT =0, C-LABOR-EST-GT, 
		C-TOTAL-EST-GT =0
		
	
OUTPUT VARIABLE DEFINITIONS
	ALPHA-NUMERIC--- O-PAINT-EST-NO, O-PAINT-YY, 
		O-PAINT-MM, O-PAINT-DD, O-PAINT-WALL-SQ-FT, 
		O-PAINT-DOOR-SQ-FT, O-TOTAL-SQ-FT, O-GALLONS-NEEDED,
		O-PAINT-PRICE-GAL, O-PAINT-EST, O-LABOR-EST, 
		O-TOTAL-EST, O-DATE, O-PCTR, O-EST-CTR, O-GAL-NEEDED-GT, 
		O-PAINT-EST-GT, O-LABOR-EST-GT, O-TOTAL-EST-GT
		
	
MAIN
	CALL INIT
	WHILE EOF = "FALSE"
		CALL MAINLINE
	CALL CLOSING
	
INIT
	OPEN INPUT FILE
	OPEN OUTPUT FILE
	REQUEST SYSTEM DATE
	GET I-DATE
	O-DATE = I-DATE
	CALL READ
	CALL HEADINGS
	
MAINLINE
	CALL CALCS
	CALL OUTPUT
	CALL READ	

CALCS
	C-TOTAL-SQ-FT = I-PAINT-WALL-SQ-FT - I-DOOR-SQ-FT
	C-GALLONS-NEEDED = C-TOTAL-SQ-FT / 115
	C-PAINT-EST = C-GALLONS-NEEDED * I-PAINT-PRICE-GAL
	C-LABOR-EST = (C-GALLONS-NEEDED * 3) *23.55
	C-TOTAL-EST = C-PAINT-EST + C-LABOR-EST
	
	ADD 1 TO C-EST-CTR
	ADD C-GALLONS-NEEDED TO C-GAL-NEEDED-GT
	ADD C-PAINT-EST TO C-PAINT-EST-GT
	ADD C-LABOR-EST TO C-LABOR-EST-GT
	ADD C-TOTAL-EST TO C-TOTAL-EST-GT

OUTPUT
	O-PAINT-EST-NO = I-PAINT-EST-NO
	O-PAINT-MM = I-PAINT-MM
	O-PAINT-DD = I-PAINT-DD
	O-PAINT-YY = I-PAINT-YY
	O-PAINT-WALL-SQ-FT = I-PAINT-WALL-SQ-FT
	O-PAINT-DOOR-SQ-FT = I-PAINT-DOOR-SQ-FT
	O-TOTAL-SQ-FT = C-TOTAL-SQ-FT
	O-GALLONS-NEEDED = C-GALLONS-NEEDED
	O-PRICE-GAL = I-PRICE-GAL
	O-PAINT-EST = C-PAINT-EST
	O-LABOR-EST = C-LABOR-EST
	O-TOTAL-EST = C-TOTAL-EST
	IF END OF PAGE
		CALL HEADINGS

CLOSING
	CALL GRAND-TOTALS
	CLOSE INPUT FILE
	CLOSE OUTPUT FILE

GRAND-TOTALS
	O-EST-CTR = C-EST-CTR
	O-GAL-NEEDED-GT = C-GAL-NEEDED-GT
	O-PAINT-EST-GT = C-PAINT-EST-GT
	O-LABOR-EST-GT = C-LABOR-EST-GT
	O-TOTAL-EST-GT = C-TOTAL-EST-GT
	PRINT BLANK LINE
	PRINT BLANK LINE
	PRINT GRAND TOTAL LINE

READ
	READ RECORD: IPAINT-EST-NO, IPAINT-DATE, I-PAINT-YY, I-PAINT-MM,
		I-PAINT-DD, I-PAINT-WALL-SQ-FT, I-PAINT-DOOR-SQ-FT, I-PAINT-PRICE-GAL
			IF END OF FILE
				EOF = "TRUE"

HEADINGS
	ADD 1 TO C-PCTR
	O-PCTR = C-PCTR
	PRINT TITLE LINE
	PRINT BLANK LINE
	PRINT COLUMN HEAGING1
	PRINT COLUMN HEADING2
	PRINT BLANK LINE
